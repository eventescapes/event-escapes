   function EventCarousel({ title, events, icon }) {
     const scrollRef = useRef(null);
     const [isPaused, setIsPaused] = useState(false);

     // Auto-scroll
     useEffect(() => {
       if (isPaused || !scrollRef.current) return;

       const interval = setInterval(() => {
         if (scrollRef.current) {
           const { scrollLeft, scrollWidth, clientWidth } = scrollRef.current;
           
           if (scrollLeft >= scrollWidth - clientWidth - 10) {
             scrollRef.current.scrollTo({ left: 0, behavior: 'smooth' });
           } else {
             scrollRef.current.scrollBy({ left: 400, behavior: 'smooth' });
           }
         }
       }, 5000);

       return () => clearInterval(interval);
     }, [isPaused]);

     const scroll = (direction) => {
       if (scrollRef.current) {

     return (

           <h2 className="text-3xl font-bold">{icon} {title}</h2>

   const fetchEventsWithClassification = async (countryCode, classificationId, classificationName) => {
     const apiKey = process.env.TICKETMASTER_API_KEY;
     const startDate = new Date();
     const endDate = new Date();
     endDate.setMonth(endDate.getMonth() + 6); // 6 months
     
     const params = new URLSearchParams({
       apikey: apiKey,
       countryCode: countryCode,
       classificationId: classificationId,
       size: '200',
       sort: 'date,asc',
       startDateTime: startDate.toISOString(),
       endDateTime: endDate.toISOString()
     });

     try {
       console.log(`Fetching ${classificationName} events for ${countryCode}...`);
       const response = await fetch(
         `https://app.ticketmaster.com/discovery/v2/events.json?${params}`
       );
       
       if (!response.ok) return [];
       
       const data = await response.json();
       const events = data._embedded?.events || [];
       console.log(`Found ${events.length} ${classificationName} events in ${countryCode}`);
       return events;
     } catch (error) {
       console.error(`Error fetching ${classificationName}:`, error);
       return [];
