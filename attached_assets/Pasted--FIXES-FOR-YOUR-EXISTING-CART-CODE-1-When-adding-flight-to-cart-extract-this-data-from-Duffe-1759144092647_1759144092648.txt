// 1. When adding flight to cart, extract this data from Duffel API response:
function addFlightToCart(flightOffer, selectedSeats) {
  const slice = flightOffer.slices[0];
  const firstSegment = slice.segments[0];
  const lastSegment = slice.segments[slice.segments.length - 1];
  
  const cartItem = {
    id: `flight-${Date.now()}`,
    type: 'flight',
    
    // Extract route info
    tripType: 'one-way', // or get from search params
    route: {
      from: firstSegment.origin.iata_code,
      to: lastSegment.destination.iata_code,
    },
    
    // Extract flight details
    airline: firstSegment.marketing_carrier.name,
    flightNumber: `${firstSegment.marketing_carrier.iata_code}${firstSegment.marketing_carrier_flight_number}`,
    departTime: new Date(firstSegment.departing_at).toLocaleTimeString('en-US', { 
      hour: '2-digit', 
      minute: '2-digit',
      hour12: false 
    }),
    arriveTime: new Date(lastSegment.arriving_at).toLocaleTimeString('en-US', { 
      hour: '2-digit', 
      minute: '2-digit',
      hour12: false 
    }),
    cabinClass: firstSegment.passengers[0].cabin_class_marketing_name || 'Economy',
    
    // Extract pricing
    baseFare: parseFloat(flightOffer.total_amount),
    currency: flightOffer.total_currency,
    
    // Seats from your seat selection modal
    seats: selectedSeats.map(seat => ({
      number: seat.designator,
      price: parseFloat(seat.total_amount)
    })),
    
    // Calculate total
    totalPrice: parseFloat(flightOffer.total_amount) + 
                selectedSeats.reduce((sum, seat) => sum + parseFloat(seat.total_amount), 0),
    
    passengers: 1, // or get from search params
    addedAt: new Date().toISOString()
  };
  
  return cartItem;
}
