import { useState, useEffect } from "react";
import { useLocation, useRoute } from "wouter";
import { Button } from "@/components/ui/button";
import { Card } from "@/components/ui/card";
import {
  Plane,
  Armchair,
  Luggage,
  CreditCard,
  ArrowLeft,
  Loader2,
} from "lucide-react";

export function BookingReviewPage() {
  const [, navigate] = useLocation();
  const [, params] = useRoute("/review/:offerId");
  const offerId = params?.offerId || "";

  const [checkoutItem, setCheckoutItem] = useState<any>(null);
  const [ancillaries, setAncillaries] = useState<any>(null);
  const [loading, setLoading] = useState(true);
  const [submitting, setSubmitting] = useState(false);

  // STEP 1: Load flight + ancillaries data
  useEffect(() => {
    const loadData = async () => {
      try {
        const local = sessionStorage.getItem("checkout_item");
        if (local) setCheckoutItem(JSON.parse(local));

        const res = await fetch(
          `${import.meta.env.VITE_SUPABASE_URL}/rest/v1/booking_ancillaries?offer_id=eq.${offerId}`,
          {
            headers: {
              apikey: import.meta.env.VITE_SUPABASE_ANON_KEY!,
              Authorization: `Bearer ${import.meta.env.VITE_SUPABASE_ANON_KEY!}`,
            },
          },
        );

        if (!res.ok) throw new Error("Failed to load ancillaries");

        const [record] = await res.json();
        if (record) setAncillaries(record);
        console.log("✅ Loaded ancillaries:", record);
      } catch (err) {
        console.error("❌ Failed to fetch review data:", err);
      } finally {
        setLoading(false);
      }
    };

    loadData();
  }, [offerId]);

  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-slate-900 via-blue-900 to-slate-900 flex items-center justify-center">
        <Loader2 className="h-12 w-12 text-blue-400 animate-spin" />
      </div>
    );
  }

  if (!checkoutItem || !checkoutItem.offer) {
    return (
      <div className="min-h-screen bg-slate-900 flex flex-col items-center justify-center text-blue-200">
        <p>Booking data missing. Please reselect your flight.</p>
        <Button
          onClick={() => navigate("/flights")}
          className="mt-4 bg-blue-500 hover:bg-blue-600 text-white"
        >
          Back to Flights
        </Button>
      </div>
    );
  }

  // Extract offer + passenger data
  const offer = checkoutItem.offer;
  const passengers = ancillaries?.passengers || checkoutItem.passengers || [];
  const seats = ancillaries?.seats || checkoutItem.selectedSeats || [];
  const baggage = ancillaries?.baggage || checkoutItem.selectedBaggage || [];
  const currency = (offer.total_currency || "AUD").toUpperCase();
  const flightPrice = parseFloat(offer.total_amount || "0");

  const seatsTotal = seats.reduce(
    (sum: number, s: any) => sum + parseFloat(s.amount || "0"),
    0,
  );
  const baggageTotal = baggage.reduce(
    (sum: number, b: any) =>
      sum + parseFloat(b.amount || "0") * (b.quantity || 1),
    0,
  );

  const extrasTotal = seatsTotal + baggageTotal;
  const grandTotal = flightPrice + extrasTotal;

  const handlePayNow = async () => {
    setSubmitting(true);
    try {
      const requestBody = {
        offerId: offer.id,
        passengers,
        services: [
          ...seats.map((s: any) => ({
            id: s.serviceId || s.id,
            type: "seat",
            amount: s.amount,
            quantity: 1,
            designator: s.designator,
          })),
          ...baggage.map((b: any) => ({
            id: b.serviceId || b.id,
            type: "baggage",
            amount: b.amount,
            quantity: b.quantity || 1,
          })),
        ],
        flightPrice: flightPrice.toFixed(2),
        seatsTotal: seatsTotal.toFixed(2),
        baggageTotal: baggageTotal.toFixed(2),
        totalAmount: grandTotal.toFixed(2),
        currency,
        offerData: offer,
      };

      const response = await fetch(
        `${import.meta.env.VITE_SUPABASE_URL}/functions/v1/create-checkout-session`,
        {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${import.meta.env.VITE_SUPABASE_ANON_KEY}`,
          },
          body: JSON.stringify(requestBody),
        },
      );

      const result = await response.json();
      if (!response.ok) throw new Error(result.error || "Checkout failed");

      console.log("✅ Redirecting to Stripe:", result.url);
      window.location.href = result.url;
    } catch (err: any) {
      alert("Payment failed: " + err.message);
      setSubmitting(false);
    }
  };

  const origin = offer.slices?.[0]?.origin?.iata_code || "N/A";
  const destination =
    offer.slices?.[offer.slices.length - 1]?.destination?.iata_code || "N/A";

  // UI Rendering
  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-blue-900 to-slate-900 py-12 px-4">
      <div className="max-w-3xl mx-auto">
        {/* Back Button */}
        <button
          onClick={() => navigate(`/passenger-details`)}
          className="flex items-center text-blue-200 hover:text-white mb-6 transition-colors"
        >
          <ArrowLeft className="w-5 h-5 mr-2" />
          Back to Passenger Details
        </button>

        {/* Header */}
        <div className="text-center mb-8">
          <h1 className="text-3xl font-bold text-white mb-2">
            Review Your Booking
          </h1>
          <p className="text-blue-200">
            Please confirm your flight details before payment
          </p>
        </div>

        {/* Flight Details */}
        <Card className="bg-white/10 backdrop-blur-lg border-white/20 p-6">
          <div className="flex items-center gap-3 mb-4">
            <Plane className="h-6 w-6 text-blue-300" />
            <h2 className="text-xl font-semibold text-white">Flight Details</h2>
          </div>
          <div className="space-y-2 text-blue-100">
            <div className="flex justify-between">
              <span className="opacity-80">Airline:</span>
              <span className="font-medium">{offer.owner?.name || "N/A"}</span>
            </div>
            <div className="flex justify-between">
              <span className="opacity-80">Route:</span>
              <span className="font-medium">
                {origin} → {destination}
              </span>
            </div>
            <div className="flex justify-between">
              <span className="opacity-80">Passengers:</span>
              <span className="font-medium">{passengers.length}</span>
            </div>
            <div className="flex justify-between">
              <span className="opacity-80">Base Flight Price:</span>
              <span className="font-medium text-lg">
                {currency} ${flightPrice.toFixed(2)}
              </span>
            </div>
          </div>
        </Card>

        {/* Seats */}
        {seats.length > 0 && (
          <Card className="bg-white/10 backdrop-blur-lg border-white/20 p-6 mt-6">
            <div className="flex items-center gap-3 mb-4">
              <Armchair className="h-6 w-6 text-blue-300" />
              <h2 className="text-xl font-semibold text-white">
                Selected Seats
              </h2>
            </div>
            <div className="space-y-3 text-blue-100">
              {seats.map((s: any, i: number) => (
                <div key={i} className="flex justify-between">
                  <span>Seat {s.designator || "N/A"}</span>
                  <span>
                    {currency} ${parseFloat(s.amount || "0").toFixed(2)}
                  </span>
                </div>
              ))}
              <div className="flex justify-between text-white font-semibold pt-2 border-t border-white/20">
                <span>Seats Subtotal:</span>
                <span>
                  {currency} ${seatsTotal.toFixed(2)}
                </span>
              </div>
            </div>
          </Card>
        )}

        {/* Baggage */}
        {baggage.length > 0 && (
          <Card className="bg-white/10 backdrop-blur-lg border-white/20 p-6 mt-6">
            <div className="flex items-center gap-3 mb-4">
              <Luggage className="h-6 w-6 text-blue-300" />
              <h2 className="text-xl font-semibold text-white">
                Selected Baggage
              </h2>
            </div>
            <div className="space-y-3 text-blue-100">
              {baggage.map((b: any, i: number) => (
                <div key={i} className="flex justify-between">
                  <span>Checked Bag × {b.quantity || 1}</span>
                  <span>
                    {currency} $
                    {(
                      parseFloat(b.amount || "0") * (b.quantity || 1)
                    ).toFixed(2)}
                  </span>
                </div>
              ))}
              <div className="flex justify-between text-white font-semibold pt-2 border-t border-white/20">
                <span>Baggage Subtotal:</span>
                <span>
                  {currency} ${baggageTotal.toFixed(2)}
                </span>
              </div>
            </div>
          </Card>
        )}

        {/* Total */}
        <Card className="bg-gradient-to-r from-blue-600/30 to-purple-600/30 backdrop-blur-lg border-white/30 p-6 mt-6">
          <div className="space-y-3">
            <div className="flex justify-between text-blue-100 text-lg">
              <span>Flight:</span>
              <span>
                {currency} ${flightPrice.toFixed(2)}
              </span>
            </div>
            {extrasTotal > 0 && (
              <div className="flex justify-between text-blue-100 text-lg">
                <span>Extras (Seats & Baggage):</span>
                <span>
                  {currency} ${extrasTotal.toFixed(2)}
                </span>
              </div>
            )}
            <div className="h-px bg-white/30 my-3" />
            <div className="flex justify-between text-white font-bold text-2xl">
              <span>Grand Total:</span>
              <span>
                {currency} ${grandTotal.toFixed(2)}
              </span>
            </div>
          </div>
        </Card>

        {/* Pay Now */}
        <Button
          onClick={handlePayNow}
          disabled={submitting}
          className="w-full bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-700 hover:to-purple-700 text-white font-semibold py-6 text-lg mt-8"
        >
          {submitting ? (
            <>
              <Loader2 className="mr-2 h-5 w-5 animate-spin" />
              Processing...
            </>
          ) : (
            <>
              <CreditCard className="mr-2 h-5 w-5" />
              Pay {currency} ${grandTotal.toFixed(2)} Now
            </>
          )}
        </Button>
        <p className="text-center text-blue-200 text-sm mt-2">
          You’ll be redirected to Stripe for secure payment
        </p>
      </div>
    </div>
  );
}
