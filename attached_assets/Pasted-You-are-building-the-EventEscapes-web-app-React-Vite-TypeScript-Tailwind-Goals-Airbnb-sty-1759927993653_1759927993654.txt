export const EDGE = import.meta.env.VITE_EDGE_URL;
export async function fetchHotels(params: Record<string,string|number>) {
  const q = new URLSearchParams(params as any).toString();
  const res = await fetch(`${EDGE}/hotelbeds-hotels?${q}`);
  if (!res.ok) throw new Error(`HB ${res.status}`);
  return res.json(); // { total, cards: [...] }
}

Hooks

Create src/hooks/useHotelSearch.ts:

import { useState } from 'react';
import { fetchHotels } from '../lib/api';
import type { HotelsResponse } from '../types';

export function useHotelSearch() {
  const [data, setData] = useState<HotelsResponse|null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string|null>(null);

  async function search(params: Record<string,string|number>) {
    setLoading(true); setError(null);
    try {
      const res = await fetchHotels(params);
      setData(res);
    } catch (e:any) {
      setError(e.message ?? 'Error');
    } finally {
      setLoading(false);
    }
  }

  return { data, loading, error, search };
}
