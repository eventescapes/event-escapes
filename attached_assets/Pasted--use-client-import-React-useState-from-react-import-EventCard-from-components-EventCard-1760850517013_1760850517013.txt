"use client";
import React, { useState } from "react";
import EventCard from "@/components/EventCard";
import LeadModal from "@/components/LeadModal";
// import your existing fetch/mapping utilities

export default function EventsPageClient({ events }: { events: any[] }) {
  const [leadOpen, setLeadOpen] = useState(false);
  const [pendingUrl, setPendingUrl] = useState<string | null>(null);
  const [pendingTitle, setPendingTitle] = useState<string | undefined>();

  function handleBook(url: string, title?: string) {
    // DO NOT navigate. Just remember the url and open the modal.
    setPendingUrl(url);
    setPendingTitle(title);
    setLeadOpen(true);
  }

  async function handleLeadSubmit(lead: { name: string; email: string }) {
    // Optional: POST to your API to save the lead (no await needed if you prefer fire-and-forget)
    try {
      fetch("/api/leads", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ ...lead, eventTitle: pendingTitle, source: "events-modal" }),
      }).catch(() => {});
    } catch {}

    // Now open Ticketmaster in a new tab/window
    if (pendingUrl) {
      window.open(pendingUrl, "_blank", "noopener,noreferrer");
    }
  }

  return (
    <>
      <LeadModal
        open={leadOpen}
        onClose={() => setLeadOpen(false)}
        onSubmit={handleLeadSubmit}
        eventTitle={pendingTitle}
      />

      <div className="grid gap-6 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 auto-rows-fr items-stretch">
        {events.map((e) => {
          const url = e.url;                // Ticketmaster URL from API
          const title = e.name;
          return (
            <div key={url + title} className="h-full">
              <EventCard
                title={title}
                dateLabel={e.dateLabel}
                venueLabel={e.venueLabel}
                imageUrl={e.imageUrl}
                priceFrom={e.priceFrom}
                currencyCode={e.currencyCode}
                onBook={() => handleBook(url, title)}
              />
            </div>
          );
        })}
      </div>
    </>
  );
}
