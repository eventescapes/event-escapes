```typescript
// Update Ticketmaster API fetch to get 3-4 months of events

const fetchEventsForRegion = async (countryCode: string) => {
  const apiKey = process.env.TICKETMASTER_API_KEY;
  
  // Calculate date range (next 4 months)
  const startDate = new Date();
  const endDate = new Date();
  endDate.setMonth(endDate.getMonth() + 4);
  
  const params = new URLSearchParams({
    apikey: apiKey,
    countryCode: countryCode,
    size: '100', // Increase to get more events
    sort: 'date,asc',
    startDateTime: startDate.toISOString(),
    endDateTime: endDate.toISOString()
  });

  try {
    const response = await fetch(
      `https://app.ticketmaster.com/discovery/v2/events.json?${params}`
    );
    
    if (!response.ok) {
      console.error(`Error fetching ${countryCode} events:`, response.status);
      return [];
    }
    
    const data = await response.json();
    return data._embedded?.events || [];
  } catch (error) {
    console.error(`Error fetching ${countryCode}:`, error);
    return [];
  }
};

// Fetch all regions sequentially
const fetchAllEvents = async () => {
  setLoading(true);
  const regions = ['US', 'CA', 'GB', 'AU'];
  const allEvents = [];
  
  for (const region of regions) {
    const events = await fetchEventsForRegion(region);
    allEvents.push(...events);
    // Small delay to avoid rate limiting
    await new Promise(resolve => setTimeout(resolve, 200));
  }
  
  // Sort by date
  const sortedEvents = allEvents.sort((a, b) => {
    const dateA = new Date(a.dates?.start?.dateTime || a.dates?.start?.localDate);
    const dateB = new Date(b.dates?.start?.dateTime || b.dates?.start?.localDate);
    return dateA - dateB;
  });
  
  setEvents(sortedEvents);
  setLoading(false);
};
```

### **2. AUTO-SCROLLING CAROUSEL COMPONENT**

```tsx
import { useState, useRef, useEffect } from 'react';
import { ChevronLeft, ChevronRight } from 'lucide-react';

function EventCarousel({ title, events, icon }) {
  const scrollContainerRef = useRef(null);
  const [canScrollLeft, setCanScrollLeft] = useState(false);
  const [canScrollRight, setCanScrollRight] = useState(true);
  const [isAutoScrolling, setIsAutoScrolling] = useState(true);

  // Check scroll position
  const checkScrollPosition = () => {
    if (scrollContainerRef.current) {
      const { scrollLeft, scrollWidth, clientWidth } = scrollContainerRef.current;
      setCanScrollLeft(scrollLeft > 0);
      setCanScrollRight(scrollLeft < scrollWidth - clientWidth - 10);
    }
  };

  // Scroll functions
  const scroll = (direction: 'left' | 'right') => {
    if (scrollContainerRef.current) {
      const scrollAmount = scrollContainerRef.current.clientWidth * 0.8;
      scrollContainerRef.current.scrollBy({
        left: direction === 'left' ? -scrollAmount : scrollAmount,
        behavior: 'smooth'
      });
    }
  };

  // Auto-scroll every 5 seconds
  useEffect(() => {
    if (!isAutoScrolling) return;

    const interval = setInterval(() => {
      if (scrollContainerRef.current) {
        const { scrollLeft, scrollWidth, clientWidth } = scrollContainerRef.current;
        
        // If at end, scroll back to start
        if (scrollLeft >= scrollWidth - clientWidth - 10) {
          scrollContainerRef.current.scrollTo({ left: 0, behavior: 'smooth' });
        } else {
          scroll('right');
        }
      }
    }, 5000);

    return () => clearInterval(interval);
  }, [isAutoScrolling]);

  return (
    <div className="mb-12">
      {/* Header */}
      <div className="flex items-center justify-between mb-6">
        <div className="flex items-center gap-3">
          <span className="text-4xl">{icon}</span>
          <h2 className="text-3xl font-bold text-white">{title}</h2>
          <span className="text-gray-400 text-sm">({events.length} events)</span>
        </div>
        
        {/* Navigation Arrows */}
        <div className="flex gap-2">
          <button
            onClick={() => scroll('left')}
            disabled={!canScrollLeft}
            className={`p-3 rounded-full bg-gray-800 hover:bg-gray-700 transition-colors ${
              !canScrollLeft ? 'opacity-50 cursor-not-allowed' : ''
            }`}
          >
            <ChevronLeft className="w-5 h-5 text-white" />
          </button>
          <button
            onClick={() => scroll('right')}
            disabled={!canScrollRight}
            className={`p-3 rounded-full bg-gray-800 hover:bg-gray-700 transition-colors ${
              !canScrollRight ? 'opacity-50 cursor-not-allowed' : ''
            }`}
          >
            <ChevronRight className="w-5 h-5 text-white" />
          </button>
        </div>
      </div>

      {/* Scrollable Container */}
      <div
        ref={scrollContainerRef}
        onScroll={checkScrollPosition}
        onMouseEnter={() => setIsAutoScrolling(false)}
        onMouseLeave={() => setIsAutoScrolling(true)}
        className="flex gap-6 overflow-x-auto scrollbar-hide scroll-smooth pb-4"
        style={{
          scrollbarWidth: 'none',
          msOverflowStyle: 'none'
        }}
      >
        {events.map((event) => (
          <div key={event.id} className="flex-shrink-0 w-80">
            <EventCard event={event} />
          </div>
        ))}
      </div>

      {/* Scroll Indicators */}
      <div className="flex justify-center gap-2 mt-4">
        {Array.from({ length: Math.ceil(events.length / 4) }).map((_, i) => (
          <div
            key={i}
            className={`h-1 rounded-full transition-all ${
              i === Math.floor((scrollContainerRef.current?.scrollLeft || 0) / (scrollContainerRef.current?.clientWidth || 1))
                ? 'w-8 bg-purple-600'
                : 'w-2 bg-gray-600'
            }`}
          />
        ))}
      </div>
    </div>
  );
}

// Add CSS to hide scrollbar
<style jsx global>{`

    display: none;
  }
```

```tsx
  });
```
