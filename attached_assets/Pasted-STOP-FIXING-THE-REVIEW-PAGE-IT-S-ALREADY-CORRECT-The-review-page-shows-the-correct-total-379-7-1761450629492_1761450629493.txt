```javascript
const createCheckoutSession = async () => {
  // Get the grand total from the review page calculation
  const grandTotal = flightPrice + seatsTotal + baggageTotal;
  
  const response = await fetch('/api/create-checkout', {
    body: JSON.stringify({
      amount: grandTotal, // ✅ CORRECT - use grand total
      line_items: [
        {
          price_data: {
            currency: 'aud',
            product_data: { name: 'Flight Booking' },
            unit_amount: Math.round(flightPrice * 100)
          },
          quantity: 1
        },
        ...(seatsTotal > 0 ? [{
          price_data: {
            currency: 'aud',
            product_data: { name: 'Seat Selection' },
            unit_amount: Math.round(seatsTotal * 100)
          },
          quantity: 1
        }] : []),
        ...(baggageTotal > 0 ? [{
          price_data: {
            currency: 'aud',
            product_data: { name: 'Baggage' },
            unit_amount: Math.round(baggageTotal * 100)
          },
          quantity: 1
        }] : [])
      ]
    })
  });
};
```
